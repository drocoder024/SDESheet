class Solution {
    public int maxSubArray(int[] nums) {
        
        int keep=nums[0];
        int max=nums[0];
        
        for(int i=1;i<nums.length;i++)
        {
            keep=Math.max(nums[i],keep+nums[i]);
            max=Math.max(keep,max);
            
        }
        return max;
    }
}
